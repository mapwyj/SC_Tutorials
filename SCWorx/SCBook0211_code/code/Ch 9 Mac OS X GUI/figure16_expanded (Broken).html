(

var createGUI, synthDefName, eventForSynthDef;

synthDefName = \default;

SynthDescLib.global.read;


//model is an Event, which can play a Synth


eventForSynthDef = {|synthDefName|

var event;

event = (

instrument: synthDefName,

play: {

var args = event.select{|it| it.isKindOf(Number)}.asKeyValuePairs;

~node = Synth(~instrument, args);

OSCpathResponder(Server.default.addr, ["/n_end", ~node.nodeID], 

{|time, resp, msg| 

~node = nil;

event.changed(\play, 0); 

resp.remove;

} 

).add;

event.changed(\play, 1);

},

stop:{

~node.release;

~node = nil;

event.changed(\play, 0);

},

setArg: {|inevent, argName, value|

inevent.use{

if(~node.notNil){

~node.set(argName, value);

};

inevent.put(argName.asSymbol, value);

inevent.changed(argName.asSymbol);

}

}

);

SynthDescLib.global[synthDefName.asSymbol].controls.do{|it|

event.put(it.name.asSymbol, it.defaultValue);

};

event

};


createGUI = {

var window, createGUIFor, synthDefListView, currentGUI, container, newSpecFor;

window = SCWindow("myWindow", Rect(100,100, 332, 320));

synthDefListView = SCPopUpMenu(window, Rect(4,4,324,20))

.items_(SynthDescLib.global.synthDescs.keys.asArray.sort)

.action_({|view|

// if there is already a composite view, remove it and reset the FlowLayout

if(currentGUI.notNil){currentGUI.remove; currentGUI=nil; window.view.decorator.reset;};

//evaluate the GUI function and store the composite view in currentGUI


currentGUI = createGUIFor.(eventForSynthDef.(view.item), window);

});


//window.view.decorator_(FlowLayout(window.view.bounds.moveTo(0,24)));

window.addFlowLayout;


createGUIFor = {|event, window|

var controlNames, playButton, height, controller, composite;


//controller

controller = SimpleController(event);


//GUI

controlNames = SynthDescLib.global[event[\instrument]].controls;

height = controlNames.size * 24 +40;

window.view.decorator.nextLine;

composite = SCScrollView(window, 324@height);

//composite.decorator = FlowLayout(composite.bounds.moveTo(0,0));

composite.decorator = FlowLayout(composite.bounds);

SCStaticText(composite, 80@20).string_(event[\instrument].asString);

playButton = SCButton(composite, 120@20)

.states_([["play"],["stop"]])

.action_({|view| 

if(view.value==1){

event.play; }{

event.stop;

}

});

//register button at controller

controller.put(\play,{|theChanger, what, value|

{playButton.value_(value)}.defer

});

composite.decorator.nextLine;

//create sliders

controlNames.do{|control, i| 

var spec, name, action, initValue, slider;

name = control.name;

spec = name.asSymbol.asSpec ? [0,1].asSpec;

action = {|view| event.setArg(name, view.value)};

initValue = control.defaultValue;

slider = EZSlider(composite, 300@20, name, spec, action, initValue);

//set a new spec on double-click on numberbox

slider.numberView.mouseDownAction_({|view, x, y, modifiers, buttonNumber, clickCount|

if(clickCount == 2){

newSpecFor.(name,slider);

}

});

composite.decorator.nextLine;

//register slider at controller

controller.put(control.asSymbol, {|theChanger, what, value|

{

slider.value_(spec.unmap(value));

}.defer

})

};

composite.onClose_{controller.remove; event.stop;};


};


window.front;


newSpecFor = {|key, slider|

var window, minbox, maxbox, warpView;

window = SCWindow("new spec for %".format(key), Rect(200,200, 348, 60));

window.view.decorator = FlowLayout(window.view.bounds);

SCStaticText(window, 80@20).string_("min:");

minbox = SCNumberBox(window, 40@20).value_(0.0);

SCStaticText(window, 80@20).string_("max:");

maxbox = SCNumberBox(window, 40@20).value_(1.0);

warpView = SCPopUpMenu(window, 80@20).items_(#[lin, exp]);

window.view.decorator.nextLine;

SCButton(window, 80@20).states_([["cancel"]]).action_({ window.close});

SCButton(window, 80@20).states_([["OK"]]).action_({ 

var spec;

spec = [minbox.value, maxbox.value, warpView.item].asSpec;

slider.controlSpec_(spec);

ControlSpec.specs.put(key, spec);

window.close;

});

window.front;

}


}.value

)
