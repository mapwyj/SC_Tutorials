//Buffer 


b = Buffer.read(s, "./sounds/Ultrabeat.aif"); // remember to free the buffer later.



//SynthDef


(

d = SynthDef(\ultrabeat, { | out = 0, bufnum = 0, loop = 1, rate = 1 |

Out.ar(out, 

PlayBuf.ar(1, bufnum, BufRateScale.kr(bufnum) * rate, loop: loop, doneAction: 2)

)

});

)


d.send(Server.default); //send Synthdef to default Server.



f = Synth(\ultrabeat, [\bufnum, b]);


/// =========================================================



//Cocoa GUI 


(

w = Window.new( "Ultrabeat Sample Slider" ).front; // create a new window


StaticText(w, Rect(0,100,400,20))

.background_(Color.rand)

.string_("Use the slider to control the rate of synth")

.align_(\center); // Put some text to the window


n = NumberBox(w, Rect(20, 20, 150, 20)); // create a numberbox 

a = Slider(w, Rect(20, 60, 150, 20)) // create a slider

.action_({

var therealvalue; // declare the variable "therealvalue"

therealvalue =  a.value - 0.5 * 5;

n.value_(therealvalue); // Numberbox value 

f.set(\rate, therealvalue);

});

)
